import matplotlib.pyplot as plt
import matplotlib.patches as patches

# Create a figure
fig, ax = plt.subplots(figsize=(10, 8))
ax.set_xlim(0, 100)
ax.set_ylim(0, 100)
plt.title("Schematic of Deep-Space Electromagnetic Propulsion System")

# Draw the main capacitor (represented as a large rectangle at the top, field directed upward)
capacitor = patches.Rectangle((30, 80), 40, 10, edgecolor='blue', facecolor='lightblue', label="Capacitor")
ax.add_patch(capacitor)
ax.annotate("Capacitor (Energy Storage)", (50, 86), color='blue', ha='center')

# Draw Tesla coil below the capacitor
tesla_coil = patches.Rectangle((25, 60), 50, 10, edgecolor='green', facecolor='lightgreen', label="Tesla Coil")
ax.add_patch(tesla_coil)
ax.annotate("Tesla Coil\n(Pre-Tuned for Deep Space)", (50, 65), color='green', ha='center')

# Draw Rotor assembly below Tesla coil
rotor_outer = patches.Circle((50, 40), 20, edgecolor='red', facecolor='none', linewidth=2, label="Rotor")
ax.add_patch(rotor_outer)
rotor_inner = patches.Circle((50, 40), 10, edgecolor='red', facecolor='none', linewidth=2)
ax.add_patch(rotor_inner)
ax.annotate("Rotor\n(Curved Coils Carry 4th Frequency)", (50, 35), color='red', ha='center')

# Draw Mercury Gas Gap between rotors - represented as a band across the rotor region
mercury_gap = patches.Rectangle((30, 38), 40, 4, edgecolor='purple', facecolor='lavender', label="Mercury Gas Gap")
ax.add_patch(mercury_gap)
ax.annotate("Mercury Gas\n(Harmonic Injection)", (50, 40), color='purple', ha='center')

# Draw Field Lines indicating upward direction (using arrows)
for x in range(40, 61, 5):
    ax.arrow(x, 80, 0, 8, head_width=1, head_length=2, fc='black', ec='black')
ax.annotate("Upward Directed Field", (50, 90), color='black', ha='center')

# Legend and labels
handles, labels = ax.get_legend_handles_labels()
ax.legend(handles, labels, loc='upper right')

plt.xlabel("System Width (arbitrary units)")
plt.ylabel("System Height (arbitrary units)")
plt.grid(True)
plt.show()
